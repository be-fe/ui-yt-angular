The $confirm service.

$confirm indicates strong warning to user. User must confirm by click ok button or close button.

$confirm options

All options can be provided by an object passing into $confirm, like $confirm({title: 'hint', body: 'good'}) :

  • title (default: 'confirm'): title of confirm.
  • body: body of confirm.

The $confirm method return a promise, which will be resolved when the confirm dismisses.

html 
          
<div ng-controller="confirmDemoCtrl">
  <h4>No Content And Default Title</h4>
  <a class="btn btn-primary" ng-click="confirm1()">confirm</a>
  <hr />
  <h4>With Content And Title</h4>
  <form class="form-horizontal">
    <div class="col-md-4">
      <input type="text" ng-model="t2.title" placeholder="title" class="form-control" />
    </div>
    <div class="col-md-4">
      <input type="text" ng-model="t2.body" placeholder="body" class="form-control" />
    </div>
    <a class="btn btn-primary" ng-click="confirm2()">confirm</a>
  </form>
  <hr />
  <h4>Confirm With Promise</h4>
  <h5 ng-if="cb">callback count: {{cb}}</h5>
  <a class="btn btn-primary" ng-click="confirm3()">confirm</a>
  <hr />
</div>
javascript 
        
angular.module('ui.yt.demo').controller('confirmDemoCtrl', function ($scope, $confirm, $log) {
  $scope.confirm1 = function () {
    $confirm();
  };
  $scope.confirm2 = function () {
    $confirm({
      title: $scope.t2.title,
      body: $scope.t2.body
    });
  };
  $scope.confirm3 = function () {
    var promise = $confirm();
    promise.then(function (result) {
      $scope.cb = result;
      $log.info('$confirm dismissed');
    });
  };
});
