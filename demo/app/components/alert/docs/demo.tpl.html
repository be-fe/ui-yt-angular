ALERT

The alert service is a subtutive of raw alert.

use $alert.pop() to show alert

html 
        
<div ng-controller="$alertDemoCtrl">
  <a class="btn btn-primary" ng-click="showAlert()">ALERT</a>
</div>
javascript 
        
angular.module('ui.yt.demo').controller('$alertDemoCtrl', function ($scope, $alert) {
  $scope.showAlert = function (argument) {
    $alert.pop();
  };
});
